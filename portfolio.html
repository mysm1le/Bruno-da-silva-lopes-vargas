Sistema de Cálculo de Planos de Saúde

            
            print("\nBem vindo ao Sistema de Bruno da Silva Lopes Vargas")
            
            # Inputs
            valorBase = float(input("Informe o valor Base do plano: R$ "))
            idade = int(input("Informe a idade do cliente: "))
            valorMensal = 0
            
            # Cálculos
            if idade >= 0 and idade < 19:
                valorMensal = valorBase * 100 / 100  # 100%
            elif idade >= 19 and idade < 29:
                valorMensal = valorBase * 150 / 100  # 150%
            elif idade >= 29 and idade < 39:
                valorMensal = valorBase * 225 / 100  # 225%
            elif idade >= 39 and idade < 49:
                valorMensal = valorBase * 240 / 100  # 240%
            elif idade >= 49 and idade < 59:
                valorMensal = valorBase * 350 / 100  # 350%
            elif idade >= 59:
                valorMensal = valorBase * 600 / 100  # 600%
            else:
                print("Idade inválida!")
            
            # Mensagens do Sistema
            print(f"Informe o valor Base do plano: R$ {valorBase:.2f}")
            print(f"Idade do cliente: {idade}")
            print(f"O valor mensal do plano é de: R$ {valorMensal:.2f}")

Pizzaria

            
            # Menu
            print("Bem-vindos à Pizzaria do Bruno da Silva Lopes Vargas!")
            print("-------------------------------------------------")
            print("------------------- Cardápio --------------------")
            print("| Tamanho  | Pizza Salgada(PS) | Pizza Doce(PD) |")
            print("| P        | R$ 30,00          | R$ 34,00       |")
            print("| M        | R$ 45,00          | R$ 48,00       |")
            print("| G        | R$ 60,00          | R$ 66,00       |")
            print("-------------------------------------------------")
            
            # Inicializando acumulador de valores
            total_pedido = 0
            
            # Loop para permitir multiplos pedidos
            while True:
                sabor = input("Entre com o sabor desejado (PS/PD): ").upper()
                if sabor not in ["PS", "PD"]:
                    print("Sabor inválido. Tente novamente")
                    continue
                tamanho = input("Entre com o tamanho desejado (P/M/G): ").upper()
                if tamanho not in ["P", "M", "G"]:
                    print("Tamanho inválido. Tente novamente")
                    continue
            
                if sabor == "PS":
                    preco = 30 if tamanho == "P" else 45 if tamanho == "M" else 60
                else:
                    preco = 34 if tamanho == "P" else 48 if tamanho == "M" else 66
            
                print(f"Você pediu uma Pizza {'Salgada' if sabor == 'PS' else 'Doce'} no tamanho {tamanho}: R$ {preco:.2f}")
                total_pedido += preco
            
                continuar = input("Deseja pedir mais alguma coisa? (S/N): ").upper()
                if continuar == "N":
                    break
            
            print(f"O valor total a ser pago: R$ {total_pedido:.2f}")

Madeireira do Lenhador

            
            print("Bem-vindos a Madeireira do Lenhador Bruno da Silva Lopes Vargas")
            print("")
            
            def escolha_tipo():
                while True:
                    print("PIN - Tora de Pinho")
                    print("PER - Tora de Peroba")
                    tipo = input("Entre com o Tipo de Madeira desejado: ").upper()
                    if tipo == "PIN":
                        return 150.40
                    elif tipo == "PER":
                        return 170.20
                    else:
                        print("Escolha inválida, tente novamente")
            
            def qtd_toras():
                while True:
                    try:
                        qtd = float(input("Quantidade de toras (m³): "))
                        if qtd < 100:
                            return qtd, 0.00
                        elif qtd < 500:
                            return qtd, 0.04
                    except ValueError:
                        print("Por favor, insira um valor válido.")
            
            if __name__ == "__main__":
                tipo = escolha_tipo()
                quantidade, desconto = qtd_toras()
                total = (tipo * quantidade) * (1 - desconto)
                print(f"Total: R$ {total:.2f}")
